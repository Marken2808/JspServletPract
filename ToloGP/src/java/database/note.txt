
CREATE TABLE STAFFS (
STAFFID INTEGER GENERATED ALWAYS AS IDENTITY NOT NULL, 
STAFFNAME VARCHAR(50) NOT NULL, 
STAFFADDRESS VARCHAR(100) NOT NULL, 
STAFFPHONE VARCHAR(20) NOT NULL,
ISAPPROVED BOOLEAN DEFAULT FALSE  NOT NULL,
SUID INTEGER NOT NULL,
PRIMARY KEY (STAFFID),
FOREIGN KEY (SUID) REFERENCES USERS(USERID));
